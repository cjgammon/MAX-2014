Masking WebGL

varying vec4 textureCoord;
uniform sampler2D mask;
uniform sampler2D map;

void main() {
	float x = (textureCoord.x / textureCoord.w);
	float y = (textureCoord.y / textureCoord.w);
	vec2 mapPosition = vec2(x, y);

	vec3 mapT = texture2D(map, mapPosition).xyz;
	vec3 maskT = texture2D(mask, mapPosition).xyz;

	float alpha = maskT.r;
	gl_FragColor = vec4(mapT, alpha);
}
